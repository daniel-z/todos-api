Create a new task [POST /api/todos]

Request [JSON]

// Requires the next data attributes
@ description: < String >
@ completed: < Boolean > 

                
        {
          description: "A new task",
          completed: false
        }

Response [JSON]

        
{
    id: 83515750,
    description: "A new task",
    completed: false
}

Update a task [PUT /api/todos/:id]

Request [JSON]

// Requires the next data attributes
@ description: < String >
@ completed: < Boolean > 

        
{
  description: "A new task",
  completed: true
}

Response [JSON]

        
{
    id: 83515750,
    description: "A new task",
    completed: true
}

Delete a task [DELETE /api/todos/:id]

Response [JSON]

        
{
    deleted: true
}
